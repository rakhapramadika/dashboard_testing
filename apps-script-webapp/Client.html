const statusBox = document.querySelector("#statusBox"); const state = { activeTab: "scorecard", scorecardGroup: "channel", }; function setStatus(message) { statusBox.textContent = message; } function formatNumber(value) { const n = Number(value || 0); return Number.isFinite(n) ? n.toLocaleString() : "-"; } function formatPct(value) { const n = Number(value); return Number.isFinite(n) ? `${(n * 100).toFixed(1)}%` : "-"; } function formatIDR(value) { const n = Number(value || 0); if (n >= 1e9) return `Rp ${(n / 1e9).toFixed(1)}B`; if (n >= 1e6) return `Rp ${(n / 1e6).toFixed(1)}M`; if (n >= 1e3) return `Rp ${(n / 1e3).toFixed(0)}K`; return `Rp ${Math.round(n).toLocaleString()}`; } function formatRate(numerator, denominator) { const den = Number(denominator || 0); if (!den) return "-"; return formatPct(Number(numerator || 0) / den); } function apiCall(action, payload = {}) { return new Promise((resolve, reject) => { google.script.run .withSuccessHandler(data => { if (data && data.error) reject(new Error(data.error)); else resolve(data); }) .withFailureHandler(error => reject(error)) .apiCall(action, payload); }); } async function refreshSummary() { setStatus("Loading summary..."); try { const filters = currentFilters(); const bundle = await apiCall("dashboard", { filters, group: state.scorecardGroup, minUsers: 50 }); const sections = ["metadata", "scorecard", "conversion", "retention", "geo"]; if (bundle.metadata && bundle.metadata.ok) hydrateFilters(bundle.metadata.data); if (bundle.scorecard && bundle.scorecard.ok) renderScorecard(bundle.scorecard.data); else renderScorecardError(bundle.scorecard && bundle.scorecard.error); if (bundle.conversion && bundle.conversion.ok) renderConversion(bundle.conversion.data); else renderTableError("#conversionBody", 8, bundle.conversion && bundle.conversion.error); if (bundle.retention && bundle.retention.ok) renderRetention(bundle.retention.data); else renderTableError("#retentionBody", 7, bundle.retention && bundle.retention.error); if (bundle.geo && bundle.geo.ok) renderGeo(bundle.geo.data); else renderTableError("#geoBody", 7, bundle.geo && bundle.geo.error); const failures = sections.filter(name => !bundle[name] || !bundle[name].ok).length; setStatus(failures ? `Summary loaded with ${failures} tab error(s).` : "Summary loaded."); } catch (error) { setStatus(error.message || String(error)); } } function currentFilters() { const channel = document.querySelector("#channelFilter").value; return { dateFrom: document.querySelector("#dateFrom").value, dateTo: document.querySelector("#dateTo").value, installChannels: channel ? [channel] : [], }; } function hydrateFilters(meta = {}) { if (meta.dateFrom) document.querySelector("#dateFrom").value = meta.dateFrom; if (meta.dateTo) document.querySelector("#dateTo").value = meta.dateTo; const channels = String(meta.installChannels || "") .split("||") .filter(Boolean); const select = document.querySelector("#channelFilter"); const selected = select.value; select.innerHTML = `` + channels.map(channel => ``).join(""); select.value = channels.includes(selected) ? selected : ""; } function renderScorecard(rows = []) { const body = document.querySelector("#scorecardBody"); const label = state.scorecardGroup === "adset" ? "Adset" : state.scorecardGroup === "store" ? "Store" : "Channel"; document.querySelector("#scorecardTitle").textContent = `Quality by ${label}`; if (!rows.length) { body.innerHTML = `No scorecard rows returned.`; updateKpis({}); return; } body.innerHTML = rows.map(row => ` ${row.groupKey || "-"} ${formatNumber(row.users)} ${formatPct(Number(row.matured) ? Number(row.repeatCount) / Number(row.matured) : null)} ${formatIDR(row.gmvPerUser)} ${formatIDR(row.aov)} ${formatPct(row.gm)} ${formatPct(row.nm)} `).join(""); const totals = rows.reduce((acc, row) => { acc.users += Number(row.users || 0); acc.registered += Number(row.matured || 0); acc.transacted += Number(row.users || 0); acc.repeat += Number(row.repeatCount || 0); return acc; }, { users: 0, registered: 0, transacted: 0, repeat: 0 }); updateKpis(totals); } function renderScorecardError(error) { document.querySelector("#scorecardBody").innerHTML = `${errorMessage(error)}`; updateKpis({}); } function renderTableError(selector, columns, error) { document.querySelector(selector).innerHTML = `${errorMessage(error)}`; } function errorMessage(error) { return String(error && error.message ? error.message : error || "Could not load data."); } function renderConversion(rows = []) { const body = document.querySelector("#conversionBody"); if (!rows.length) { body.innerHTML = `No conversion rows returned.`; return; } body.innerHTML = rows.map(row => ` ${row.channel || "-"} ${formatNumber(row.installs)} ${formatNumber(row.registered)} ${formatNumber(row.transacted)} ${formatPct(row.registerRate)} ${formatPct(row.transactionRate)} ${formatPct(row.repeatRate)} ${formatNumber(row.uninstalled)} `).join(""); } function renderRetention(rows = []) { const body = document.querySelector("#retentionBody"); if (!rows.length) { body.innerHTML = `No retention rows returned.`; return; } body.innerHTML = rows.map(row => ` ${row.groupKey || "-"} ${row.cohortMonth || "-"} ${formatNumber(row.n)} ${formatRate(row.d30, row.n)} ${formatRate(row.r60, row.m60)} ${formatRate(row.r90, row.m90)} ${formatRate(row.r91, row.m91)} `).join(""); } function renderGeo(rows = []) { const body = document.querySelector("#geoBody"); const stores = rows.filter(row => row.rowType === "store"); if (!stores.length) { body.innerHTML = `No geo rows returned.`; return; } body.innerHTML = stores.map(row => ` ${row.store || "-"} ${formatNumber(row.users)} ${formatNumber(row.withCoords)} ${formatRate(row.withCoords, row.users)} ${formatIDR(row.totalGmv)} ${row.store_latitude || "-"} ${row.store_longitude || "-"} `).join(""); } function updateKpis(totals) { document.querySelector("#kpiInstalls").textContent = formatNumber(totals.users); document.querySelector("#kpiRegistered").textContent = formatNumber(totals.registered);
